// Common Mastermind
ACTOR CommonMindRPG : CommonMind Replaces CommonMind
{
	Tag "Spider Mastermind"

	States
	{
	Spawn:
		SPID A 0 NoDelay ACS_NamedExecuteAlways("MonsterInit", 0, MF_BOSS)
		Goto Super::Spawn
	See:
		SPID A 0 ACS_NamedExecuteAlways("MonsterInit", 0, MF_BOSS)
		Goto Super::See
	}
}
