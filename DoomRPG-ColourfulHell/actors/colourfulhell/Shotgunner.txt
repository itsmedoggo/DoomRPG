// Common Shotgunner
ACTOR CommonSGRPG : CommonSG Replaces CommonSG
{
	Tag "Former Sergeant"

	States
	{
	Spawn:
		SPOS A 0 NoDelay ACS_NamedExecuteAlways("MonsterInit", 0)
		Goto Super::Spawn
	See:
		SPOS A 0 ACS_NamedExecuteAlways("MonsterInit", 0)
		Goto Super::See
	}
}
