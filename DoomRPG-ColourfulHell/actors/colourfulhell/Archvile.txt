// Common Archvile
ACTOR CommonArchRPG : CommonArch Replaces CommonArch
{
	Tag "Archvile"

	States
	{
	Spawn:
		VILE A 0 NoDelay ACS_NamedExecuteAlways("MonsterInit", 0)
		Goto Super::Spawn
	See:
		VILE A 0 ACS_NamedExecuteAlways("MonsterInit", 0)
		Goto Super::See
	}
}
